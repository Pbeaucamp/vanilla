    <mx:Script>
        <![CDATA[
            import org.un.cava.birdeye.ravis.utils.LogUtil;
            import org.un.cava.birdeye.ravis.utils.events.VGraphEvent;
            import org.un.cava.birdeye.ravis.graphLayout.visual.VisualGraph;
            import org.un.cava.birdeye.ravis.enhancedGraphLayout.event.VGNodeEvent;
            import org.un.cava.birdeye.ravis.utils.events.VNodeMouseEvent;
              
            private var _vgraph:VisualGraph;
            /**
             * Provides access to the registered vgraph object.
             * */
            public function set vgraph(v:VisualGraph):void {
                _vgraph = v;
                registerListeners();
            }

            /**
             * @private
             * */
            public function get vgraph():VisualGraph {
                return _vgraph;
            }
            
            /**
             * register listeners
             * */
            private function registerListeners():void {
                _vgraph.addEventListener(VGNodeEvent.VG_NODE_CLICK, refreshNodeData);
            }
            private function refreshNodeData(event:VGNodeEvent):void{
                if (event.node.data is XML){
                    this.nodeName.text = event.node.data.@name;
                    var nodeClass:String = event.node.data.@dataClass;
                    var userClass:String = "bpm.birep.admin.datas.vanilla.User";
                    var itemClass:String = "bpm.repository.api.model.Item";
                    trace(nodeClass);
                    trace(userClass);
                    if (nodeClass == userClass){
                        bMail.text = "Buisness Mail : " +event.node.data.@bMail;
                        cellular.text = "Cellular : "+event.node.data.@cellular;
                        fonction.text = "Function : "+event.node.data.@fonction;
                        login.text = "Login : "+event.node.data.@login;
                        pMail.text = "Private Mail : "+event.node.data.@pMail;
                        sName.text = "Skype Name : "+event.node.data.@sName;
                        phone.text = "Phone : "+event.node.data.@phone;
                        sNumber.text = "Skype Phone : "+event.node.data.@sNumber;
                        this.currentState ="UserDisplay";
                    }else if (nodeClass == itemClass){
                        creation.text = "creation : "+event.node.data.@creation;
                        creator.text = "creator : "+event.node.data.@creator;
                        itemId.text = "item ID : "+event.node.data.@itemId;
                        modification.text = "Modification : "+event.node.data.@modification;
                        modifier.text = "Modifier : "+event.node.data.@modifier;
                        type.text = "Type : "+event.node.data.@type;
                        this.currentState = "ItemDisplay";
                    }
                    else{
                        this.currentState="";
                    }
                }    
            }
