                    break;

                case "fasd_stop":

                    break;
                case "fasd_run":
                    img.source = EmbeddedIcons.fasd_run;
                    break;
                case "fwr":
                    img.source = EmbeddedIcons.fwr;
                    break;
                case "fd_deployed_run":
                    img.source = EmbeddedIcons.fd_deployed_run;
                    break;
                case "fd_deployed_stop":
